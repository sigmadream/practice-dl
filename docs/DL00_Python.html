def bubble_sort(x):
  arr = x.copy()
  end = len(arr) - 1
  while end > 0:
    last_swap = 0
    for i in range(end):
      if arr[i] > arr[i + 1]:
        arr[i], arr[i + 1] = arr[i + 1], arr[i]
        last_swap = i
    end = last_swap
  return arr

class BubbleSort():
  def __init__(self, x):
    self.arr = x.copy()

  def sort(self):
    end = len(self.arr) - 1
    while end > 0:
      last_swap = 0
      for i in range(end):
        if self.arr[i] > self.arr[i + 1]:
          self.arr[i], self.arr[i + 1] = self.arr[i + 1], self.arr[i]
          last_swap = i
      end = last_swap
    return self.arr
bubble_sort = BubbleSort([5,3,2,1,3,2,1])

bubble_sort <- function(x) {
  n <- length(x)
  for (i in 1:(n-1)) {
    for (j in 1:(n-i)) {
      if (x[j] > x[j+1]) {
        temp <- x[j]
        x[j] <- x[j+1]
        x[j+1] <- temp
      }
    }
  }
  return(x)
}

bubble_sort <- function(x) {
  swapped <- TRUE
  while (swapped) {
    swapped <- FALSE
    for (i in 1:(length(x) - 1)) {
      if (x[i] > x[i + 1]) {
        tmp <- x[i]
        x[i] <- x[i + 1]
        x[i + 1] <- tmp
        swapped <- TRUE
      }
    }
  }
  return(x)
}
